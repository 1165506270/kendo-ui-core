Binding to Remote Data
                    
HTML

                    <input id="titles" />

                    
Script

                    $("titles").kendoAutoComplete({
                        dataTextField: "Name",
                        dataSource: new kendo.data.DataSource({
                            type: "odata",
                            serverPaging: true,
                            pageSize: 20,
                            transport: {
                                read: {
                                    url: "http://odata.netflix.com/Catalog/Titles",
                                    data: { //additional parameters sent to the remote service
                                        q: function() {
                                            return "$filter=substringof('" + $("#titles").val() + "',Name)";
                                        }
                                    }
                                }
                            }
                        })
                    });
