This example shows how to attach events to a kendoTabStrip and how they are fired and triggered.
                $("#input").kendoDropDownList({
                    dataSource: data,
                    change: function(e) { kendoConsole.log("Changed: {text: " + this.text() + " , value: " + this.value() + "}"); },
                    close: function(e) { kendoConsole.log("List is closing"); },
                    open: function(e) { kendoConsole.log("List is opening"); }
                });
